═══════════════════════════════════════

1. ADD NEW TABLES (don't modify existing):

CREATE TABLE exam_categories (
  id INT PRIMARY KEY,
  name VARCHAR(100), -- UPSC, SSC, BANKING, RAILWAY, STATE_PSC, DEFENSE, SCHOOL_OLYMPIAD
  description TEXT,
  icon VARCHAR(50),
  active BOOLEAN DEFAULT true
);

CREATE TABLE competitive_exams (
  id INT PRIMARY KEY,
  category_id INT REFERENCES exam_categories(id),
  name VARCHAR(200), -- e.g., "UPSC Civil Services Preliminary 2026"
  exam_type VARCHAR(50), -- IAS, SSC_CGL, IBPS_PO, etc.
  target_audience VARCHAR(50), -- GRADUATION, POST_GRADUATION
  syllabus JSON, -- Sections: Reasoning, Quant, English, GK, Computer
  exam_date DATE,
  registration_fee DECIMAL,
  total_marks INT,
  duration_minutes INT,
  negative_marking BOOLEAN,
  exam_mode VARCHAR(20) -- ONLINE, OFFLINE, HYBRID
);

2. EXTEND existing olympiads table:
ALTER TABLE olympiads ADD COLUMN exam_category VARCHAR(50) DEFAULT 'SCHOOL_OLYMPIAD';
ALTER TABLE olympiads ADD COLUMN target_level VARCHAR(50); -- Class 1-12, GRADUATION, POST_GRADUATION

3. EXTEND users table:
ALTER TABLE users ADD COLUMN user_type VARCHAR(20) DEFAULT 'STUDENT'; -- STUDENT, ASPIRANT
ALTER TABLE users ADD COLUMN education_level VARCHAR(50); -- For aspirants: GRADUATION, POST_GRADUATION
ALTER TABLE users ADD COLUMN target_exam_category VARCHAR(50); -- UPSC, SSC, BANKING, etc.
